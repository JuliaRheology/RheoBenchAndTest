using Revise; using PyPlot; using DSP; using SpecialFunctions;

function derivBD(y::Array{T,1}, x::Array{T,1}) where T<:Real

    # assert y and x arrays are same length
    @assert length(y)==length(x) "X and Y Array lengths must match."

    # initialise zero array of length y
    ydot = similar(y)
    @inbounds for i in eachindex(y)
        if i==start(eachindex(y))
            # assume 'imaginary' previous point is 0.0, and Δx is the same as the next one ahead
            # this is a physical assumption that material is at rest before first data point.
            ydot[i] = (y[1] - 0.0)/(x[2] - x[1]) 
        else
            # backward difference for rest of array
            ydot[i] = (y[i] - y[i-1])/(x[i] - x[i-1])
        end
    end
    ydot

    end;

function boltzconvolve(modulus::Function, time_series::Array{Float64,1}, dt::Float64,
                        params::Array{Float64,1}, prescribed_dot::Array{Float64,1})::Array{Float64,1}

    Modulus = modulus(time_series, params)
    # fast convolution
    β = conv(Modulus, prescribed_dot)
    # pick out relevant elements (1st half) and multiply by dt
    β = β[1:length(time_series)]*dt

end;

function G_springpot(t::Array{T,1}, params::Array{T,1}) where T<:Real
    β, = params
    
    G = t.^(-β)/gamma(1 - β)
end;

function analytic_con_t2(t, β)
    #loading: 0.5t^2
    return t.^(2-β)/gamma(3-β) 
end;    

function analytic_parabola(t, β)
    #loading: 2500 - (t - 50)^2 (inverse parabola which is 0 at t=0 and maximum at t=50.)
    return (100/gamma(2-β))*t.^(1-β)  .-  (2/gamma(3-β))*t.^(2-β)
end;

dt = 1e-2
t = collect(0.0:dt:1e2);
tlen = length(t);

β = 0.2
param = [β]

ϵ = [0.5*i^2 for i in t]
ϵdot = [i for i in t] # know this exactly

exact = analytic_con_t2(t, β);

guesses = collect(1e-2:1e-3/3.0:2e-1) # need higher bound for lower β !!! What's going on!
errors = similar(guesses)
for (i,v) in enumerate(guesses)
    t_mod = vcat([v], t[2:end])
    convolved = boltzconvolve(G_springpot, t_mod, dt, param, ϵdot);
    errors[i] = sum((convolved - exact).^2);
end

t_found = guesses[argmin(errors)]
println("Optimum approximation: $t_found")

lo_guess = 2e-2
hi_guess = 1e-1

fig, ax = subplots(1, 1, figsize=(6, 4.5))
ax[:loglog](guesses, errors, label="");
ax[:loglog]([lo_guess, lo_guess], [minimum(errors), maximum(errors)], "--", color="green", label="");
ax[:loglog]([hi_guess, hi_guess], [minimum(errors), maximum(errors)], "--", color="red", label="");
ax[:loglog]([t_found, t_found], [minimum(errors), maximum(errors)], "-.", color = "black", alpha=0.5, label="");
# ax[:legend](loc="best")
ax[:set_xlabel]("Singularity Approximation Point (s)")
ax[:set_ylabel]("Sum of Squared Error");
